import pygame import random import sys # 1. إعداد المكتبة والنافذة pygame.init() WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Urban Legacy 2D - Mini GTA") # 2. الألوان COLOR_BG = (50, 50, 50) # لون الشارع COLOR_BUILDING = (90, 90, 100) # لون المباني COLOR_PLAYER = (0, 200, 255) # لون سيارتك (أزرق) COLOR_MONEY = (0, 255, 100) # الفلوس (أخضر) COLOR_POLICE = (255, 50, 50) # الشرطة (أحمر) COLOR_TEXT = (255, 255, 255) clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 22, bold=True) # 3. متغيرات اللاعب player = pygame.Rect(400, 300, 25, 25) speed = 5 money_score = 0 health = 100 wanted_level = 0 # 4. المباني (عقبات لا يمكنك اختراقها) buildings = [ pygame.Rect(100, 100, 180, 140), pygame.Rect(520, 100, 180, 140), pygame.Rect(100, 360, 180, 140), pygame.Rect(520, 360, 180, 140), ] # 5. الفلوس والشرطة money_bags = [pygame.Rect(random.randint(50, WIDTH-50), random.randint(50, HEIGHT-50), 15, 15) for _ in range(6)] police_cars = [] def spawn_police(): return pygame.Rect(random.choice([20, WIDTH-40]), random.choice([20, HEIGHT-40]), 25, 25) # الحلقة الرئيسية للعبة (Game Loop) running = True while running: clock.tick(60) # 60 إطار في الثانية for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # التحكم بالحركة (WASD أو الأسهم) keys = pygame.key.get_pressed() dx, dy = 0, 0 if keys[pygame.K_LEFT] or keys[pygame.K_a]: dx -= speed if keys[pygame.K_RIGHT] or keys[pygame.K_d]: dx += speed if keys[pygame.K_UP] or keys[pygame.K_w]: dy -= speed if keys[pygame.K_DOWN] or keys[pygame.K_s]: dy += speed # اصطدام اللاعب بالمباني افقياً ورأسياً player.x += dx for b in buildings: if player.colliderect(b): if dx > 0: player.right = b.left if dx < 0: player.left = b.right player.y += dy for b in buildings: if player.colliderect(b): if dy > 0: player.bottom = b.top if dy < 0: player.top = b.bottom # منع الخروج من حدود الشاشة player.clamp_ip(pygame.Rect(0, 0, WIDTH, HEIGHT)) # جمع الفلوس for m in money_bags[:]: if player.colliderect(m): money_bags.remove(m) money_score += 100 money_bags.append(pygame.Rect(random.randint(50, WIDTH-50), random.randint(50, HEIGHT-50), 15, 15)) # زيادة النجوم والشرطة كل ما جمعت فلوس أكثر if money_score % 300 == 0 and wanted_level < 5: wanted_level += 1 police_cars.append(spawn_police()) # ذكاء اصطناعي للشرطة (تلاحق مكان اللاعب) for p in police_cars: if p.x < player.x: p.x += 2 if p.x > player.x: p.x -= 2 if p.y < player.y: p.y += 2 if p.y > player.y: p.y -= 2 # عند اصطدام الشرطة باللاعب if player.colliderect(p): health -= 1 # --- الرسم على الشاشة --- screen.fill(COLOR_BG) # رسم المباني for b in buildings: pygame.draw.rect(screen, COLOR_BUILDING, b) # رسم الفلوس for m in money_bags: pygame.draw.rect(screen, COLOR_MONEY, m) # رسم الشرطة for p in police_cars: pygame.draw.rect(screen, COLOR_POLICE, p) # رسم اللاعب pygame.draw.rect(screen, COLOR_PLAYER, player) # عرض الواجهة (HUD) stars_display = "★" * wanted_level hud = font.render(f"Money: ${money_score} | Health: {max(0, health)}% | Wanted: {stars_display}", True, COLOR_TEXT) screen.blit(hud, (15, 15)) # نهاية اللعبة if health <= 0: over_text = font.render("WASTED - GAME OVER", True, (255, 50, 50)) screen.blit(over_text, (WIDTH//2 - 120, HEIGHT//2)) pygame.display.flip() pygame.time.wait(3000) running = False pygame.display.flip() pygame.quit() sys.exit()
